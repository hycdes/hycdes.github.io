I/O方法

ios::fixed保留N位小数

#include<iostream>

#include<iomanip>

float f = 1.2;

...

cout.setf(ios::fixed);  //设置按位补0

cout << setprecision(2) << f;  //输出1.20

cout.unsetf(ios::fixed);  //取消按位补0

cout << setprecision(2) << f;  //输出1.2

I/O:iomanip
setfill输出固定位整数

#include<iostream>

#include<iomanip>

int n = 1, m = 123456, s = -1;

...

cout << setw(5) << setfill('0') << n;  //输出00001

cout << setw(5) << setfill('0') << m;  //超过位数如实输出123456

cout << setw(5) << setfill('0') << s;  //输出000-1

fstream开关文件

#include<fstream>

ifstream ifs;  //输入流

ofstream ofs;  //输出流

ifs.open("file");  //读取当前目录名为file的文件

ifs.close();  //关闭文件

is_open()属性 检查文件是否打开

可以设置一个ifs_test专门测试文件是否存在：

ifstream ifs_test, ifs_file;

string str_filename;

while(true){

    cin >> str_filename;

    ifs_test.open(str_filename);

    if(!ifs_test.is_open()){

        cout << "File not exist, try again." << endl;

        ifs_test.close();

    }else{

        ifs_test.close();

        break;

    }

}
